<body>

    <div id="board"></div>

    <div id="controls">

        <div id="dice" onclick="rollDice()">?</div>

        <p>当前玩家：<span id="current-player">红</span></p>

        <p>按回车键或点击骰子开始</p>

    </div>


    <script>

        const players = [

            { color: 'red', position: 0, element: null },

            { color: 'blue', position: 0, element: null },

            { color: 'green', position: 0, element: null },

            { color: 'yellow', position: 0, element: null }

        ];

        

        let currentPlayer = 0;

        let isRolling = false;

        const cells = [];

        const CELL_COUNT = 20;

        const PLAYER_COLORS = ['red', 'blue', 'green', 'yellow'];


        // 初始化棋盘

        function initBoard() {

            const board = document.getElementById('board');

            

            // 创建棋盘格子

            for (let i = 0; i < CELL_COUNT; i++) {

                const cell = document.createElement('div');

                cell.className = 'cell';

                cell.textContent = i + 1;

                

                // 计算位置（简单环形布局）

                const angle = (i * 360 / CELL_COUNT) * Math.PI / 180;

                const x = 300 + 250 * Math.cos(angle) - 30;

                const y = 300 + 250 * Math.sin(angle) - 30;

                

                cell.style.left = x + 'px';

                cell.style.top = y + 'px';

                board.appendChild(cell);

                cells.push({x, y});

            }


            // 创建玩家棋子

            players.forEach((player, index) => {

                const element = document.createElement('div');

                element.className = 'player';

                element.style.backgroundColor = player.color;

                element.style.left = (cells[0].x + 20 * (index % 2)) + 'px';

                element.style.top = (cells[0].y + 20 * Math.floor(index / 2)) + 'px';

                board.appendChild(element);

                player.element = element;

            });

        }


        // 掷骰子逻辑

        function rollDice() {

            if (isRolling) return;

            

            isRolling = true;

            const dice = document.getElementById('dice');

            dice.classList.add('dice-roll');

            

            setTimeout(() => {

                const result = Math.floor(Math.random() * 6) + 1;

                dice.textContent = result;

                dice.classList.remove('dice-roll');

                movePlayer(result);

                isRolling = false;

            }, 500);

        }


        // 移动玩家

        function movePlayer(steps) {

            const player = players[currentPlayer];

            player.position = (player.position + steps) % CELL_COUNT;

            

            const targetCell = cells[player.position];

            const offsetX = 20 * (currentPlayer % 2);

            const offsetY = 20 * Math.floor(currentPlayer / 2);

            

            player.element.style.left = (targetCell.x + offsetX) + 'px';

            player.element.style.top = (targetCell.y + offsetY) + 'px';

            

            // 切换到下一个玩家

            currentPlayer = (currentPlayer + 1) % players.length;

            document.getElementById('current-player').textContent = 

                PLAYER_COLORS[currentPlayer];

        }


        // 键盘事件监听

        document.addEventListener('keydown', (e) => {

            if (e.key === 'Enter') rollDice();

        });


        // 初始化游戏

        initBoard();

    </script>
